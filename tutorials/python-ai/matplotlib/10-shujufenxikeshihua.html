数据分析可视化

掌握数据分析中的可视化技巧

📚 学习目标

  • 掌握分布图绘制
  • 学会相关性可视化
  • 掌握分组可视化
  • 学会综合分析图表

1. 分布图

import matplotlib.pyplot as plt
import numpy as np

data = np.random.randn(1000)

fig, axes = plt.subplots(1, 3, figsize=(15, 5))

# 直方图
axes[0].hist(data, bins=30, edgecolor='black')
axes[0].set_title('Histogram')

# KDE密度图
from scipy.stats import gaussian_kde
density = gaussian_kde(data)
xs = np.linspace(data.min(), data.max(), 200)
axes[1].plot(xs, density(xs))
axes[1].fill_between(xs, density(xs), alpha=0.3)
axes[1].set_title('KDE Plot')

# 箱线图
axes[2].boxplot(data, vert=True)
axes[2].set_title('Box Plot')

plt.tight_layout()
plt.show()

2. 相关性矩阵

import pandas as pd

# 创建数据
data = pd.DataFrame(np.random.randn(100, 4), 
                   columns=['A', 'B', 'C', 'D'])
corr = data.corr()

fig, ax = plt.subplots(figsize=(8, 6))
im = ax.imshow(corr, cmap='coolwarm', aspect='auto', vmin=-1, vmax=1)

# 添加数值标签
for i in range(len(corr)):
    for j in range(len(corr)):
        text = ax.text(j, i, f'{corr.iloc[i, j]:.2f}',
                      ha='center', va='center', color='black')

ax.set_xticks(range(len(corr.columns)))
ax.set_yticks(range(len(corr.columns)))
ax.set_xticklabels(corr.columns)
ax.set_yticklabels(corr.columns)
plt.colorbar(im, ax=ax)
ax.set_title('Correlation Matrix')
plt.tight_layout()
plt.show()

3. 分组可视化

categories = ['A', 'B', 'C']
data = {cat: np.random.randn(100) + i for i, cat in enumerate(categories)}

fig, ax = plt.subplots(figsize=(10, 6))
positions = range(1, len(categories) + 1)
ax.boxplot(data.values(), positions=positions, labels=categories)
ax.set_title('Grouped Box Plot')
ax.set_xlabel('Category')
ax.set_ylabel('Value')
plt.show()
